a = int(input("First_value : "))
b = int(input("Second_value : "))
o = str(input("Enter the operator : "))

if o == "+" :
    c = a + b 
    print("answer: " + str(a) + " + " + str(b) + " = " + str(c))
elif o == "-" :
    c = a - b
    print("answer: " + str(a) + " - " + str(b) + " = " + str(c))
elif o == "*" :
    c = a * b 
    print("answer: " + str(a) + " * " + str(b) + " = " + str(c))
elif o == "/" :
    c = a / b 
    print("answer: " + str(a) + " / " + str(b) + " = " + str(c))
